1. Basic Trading Strategy

Implementing a simple moving average crossover strategy.

import pandas as pd
import numpy as np
import yfinance as yf

def moving_average_strategy(symbol, short_window=50, long_window=200):
    # Download data
    df = yf.download(symbol, start='2022-01-01', end='2023-12-31')
    
    # Calculate moving averages
    df['Short_MA'] = df['Close'].rolling(window=short_window).mean()
    df['Long_MA'] = df['Close'].rolling(window=long_window).mean()
    
    # Generate signals
    df['Signal'] = 0
    df['Signal'] = np.where(df['Short_MA'] > df['Long_MA'], 1, 0)
    df['Position'] = df['Signal'].diff()
    
    # Calculate returns
    df['Returns'] = df['Close'].pct_change()
    df['Strategy_Returns'] = df['Returns'] * df['Signal'].shift(1)
    
    return df

# Test the strategy
aapl_signals = moving_average_strategy('AAPL')
cumulative_returns = (1 + aapl_signals['Strategy_Returns']).cumprod()

2. Portfolio Optimization

Implementing Modern Portfolio Theory (MPT) for portfolio optimization.

def optimize_portfolio(symbols, start_date, end_date):
    # Download data
    data = yf.download(symbols, start=start_date, end=end_date)['Adj Close']
    
    # Calculate returns
    returns = data.pct_change()
    
    # Calculate mean returns and covariance
    mean_returns = returns.mean()
    cov_matrix = returns.cov()
    
    # Generate random portfolio weights
    weights = np.random.random(len(symbols))
    weights = weights / np.sum(weights)
    
    # Calculate portfolio metrics
    portfolio_return = np.sum(mean_returns * weights) * 252
    portfolio_std = np.sqrt(np.dot(weights.T, np.dot(cov_matrix * 252, weights)))
    
    return {
        'Return': portfolio_return,
        'Risk': portfolio_std,
        'Sharpe': portfolio_return / portfolio_std,
        'Weights': dict(zip(symbols, weights))
    }

3. Risk Management

Implementing risk management techniques.

def calculate_risk_metrics(returns):
    # Value at Risk (VaR)
    var_95 = np.percentile(returns, 5)
    
    # Expected Shortfall (CVaR)
    cvar_95 = returns[returns <= var_95].mean()
    
    # Maximum Drawdown
    cumulative = (1 + returns).cumprod()
    running_max = np.maximum.accumulate(cumulative)
    drawdowns = (cumulative - running_max) / running_max
    max_drawdown = np.min(drawdowns)
    
    return {
        'VaR_95': var_95,
        'CVaR_95': cvar_95,
        'Max_Drawdown': max_drawdown,
        'Volatility': returns.std() * np.sqrt(252),
        'Sharpe_Ratio': returns.mean() / returns.std() * np.sqrt(252)
    }

      # Example momentum strategy implementation
import yfinance as yf
import pandas as pd

def simple_momentum_strategy(symbols, lookback_period=12):
    # Download data
    data = yf.download(symbols, 
                      start='2022-01-01', 
                      end='2023-12-31')['Adj Close']
    
    # Calculate monthly returns
    monthly_returns = data.resample('M').last().pct_change()
    
    # Calculate momentum score (12-month return)
    momentum_score = monthly_returns.rolling(window=lookback_period).sum()
    
    # Rank stocks by momentum
    ranked_stocks = momentum_score.rank(axis=1, ascending=False)
    
    # Create signal (1 for top 3 stocks, 0 for others)
    signals = (ranked_stocks <= 3).astype(int)
    
    return signals

# Example usage:
symbols = ['AAPL', 'MSFT', 'GOOGL', 'AMZN', 'META']
signals = simple_momentum_strategy(symbols)
