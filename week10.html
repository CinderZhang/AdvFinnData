2. Main Application

Creating the main application file (main.py).

import yfinance as yf
import pandas as pd
import matplotlib.pyplot as plt
from src.data_loader import load_stock_data
from src.analysis import analyze_stock
from src.visualization import create_dashboard

def main():
    try:
        # Get user input
        symbol = input("Enter stock symbol (e.g., AAPL): ").upper()
        start_date = input("Enter start date (YYYY-MM-DD): ")
        end_date = input("Enter end date (YYYY-MM-DD): ")
        
        # Load data
        stock_data = load_stock_data(symbol, start_date, end_date)
        
        # Perform analysis
        analysis_results = analyze_stock(stock_data)
        
        # Create visualization
        create_dashboard(stock_data, analysis_results)
        
        # Print summary
        print("\nAnalysis Summary:")
        print("-" * 40)
        for metric, value in analysis_results.items():
            print(f"{metric}: {value:.4f}")
            
    except Exception as e:
        print(f"Error: {str(e)}")
        
if __name__ == "__main__":
    main()

3. Data Loading Module

Creating the data loading module (data_loader.py).

import yfinance as yf
import pandas as pd

def load_stock_data(symbol, start_date, end_date):
    """
    Load stock data from Yahoo Finance.
    
    Parameters:
        symbol (str): Stock symbol
        start_date (str): Start date in YYYY-MM-DD format
        end_date (str): End date in YYYY-MM-DD format
        
    Returns:
        pd.DataFrame: Stock data
    """
    try:
        # Download data
        data = yf.download(symbol, start=start_date, end=end_date)
        
        if data.empty:
            raise ValueError(f"No data found for {symbol}")
            
        return data
        
    except Exception as e:
        raise Exception(f"Error loading data: {str(e)}")

4. Analysis Module

Creating the analysis module (analysis.py).

import numpy as np
import pandas as pd

def analyze_stock(data):
    """
    Perform stock analysis.
    
    Parameters:
        data (pd.DataFrame): Stock data
        
    Returns:
        dict: Analysis results
    """
    # Calculate returns
    data['Returns'] = data['Adj Close'].pct_change()
    
    # Calculate metrics
    total_return = (data['Adj Close'][-1] / data['Adj Close'][0]) - 1
    volatility = data['Returns'].std() * np.sqrt(252)
    sharpe_ratio = data['Returns'].mean() / data['Returns'].std() * np.sqrt(252)
    
    # Calculate drawdown
    rolling_max = data['Adj Close'].expanding().max()
    drawdown = (data['Adj Close'] - rolling_max) / rolling_max
    max_drawdown = drawdown.min()
    
    return {
        'Total Return': total_return,
        'Annual Volatility': volatility,
        'Sharpe Ratio': sharpe_ratio,
        'Maximum Drawdown': max_drawdown
    }
