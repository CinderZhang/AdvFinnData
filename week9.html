1. Stock Analysis Dashboard

Creating a comprehensive stock analysis tool.

import yfinance as yf
import pandas as pd
import matplotlib.pyplot as plt

def create_stock_dashboard(symbol, start_date, end_date):
    # Download data
    stock = yf.download(symbol, start=start_date, end=end_date)
    
    # Calculate metrics
    stock['Daily_Return'] = stock['Adj Close'].pct_change()
    stock['MA50'] = stock['Adj Close'].rolling(window=50).mean()
    stock['MA200'] = stock['Adj Close'].rolling(window=200).mean()
    
    # Create subplots
    fig, (ax1, ax2, ax3) = plt.subplots(3, 1, figsize=(12, 10))
    
    # Price and MA plot
    ax1.plot(stock.index, stock['Adj Close'], label='Price')
    ax1.plot(stock.index, stock['MA50'], label='50-day MA')
    ax1.plot(stock.index, stock['MA200'], label='200-day MA')
    ax1.set_title(f'{symbol} Stock Price and Moving Averages')
    ax1.legend()
    
    # Volume plot
    ax2.bar(stock.index, stock['Volume'])
    ax2.set_title('Trading Volume')
    
    # Returns distribution
    ax3.hist(stock['Daily_Return'].dropna(), bins=50)
    ax3.set_title('Daily Returns Distribution')
    
    plt.tight_layout()
    plt.show()
    
    # Calculate summary statistics
    summary = {
        'Average Daily Return': stock['Daily_Return'].mean(),
        'Return Volatility': stock['Daily_Return'].std(),
        'Total Return': (stock['Adj Close'][-1] / stock['Adj Close'][0]) - 1,
        'Max Drawdown': (stock['Adj Close'] / stock['Adj Close'].expanding().max() - 1).min()
    }
    
    return pd.Series(summary)

2. Portfolio Analysis Tool

Building a tool to analyze portfolio performance.

def analyze_portfolio(symbols, weights, start_date, end_date):
    # Download data for all symbols
    portfolio = pd.DataFrame()
    for symbol in symbols:
        data = yf.download(symbol, start=start_date, end=end_date)['Adj Close']
        portfolio[symbol] = data
    
    # Calculate portfolio value
    normalized = portfolio / portfolio.iloc[0]
    weighted_returns = normalized * weights
    portfolio_value = weighted_returns.sum(axis=1)
    
    # Calculate metrics
    daily_returns = portfolio_value.pct_change()
    
    metrics = {
        'Total Return': (portfolio_value[-1] / portfolio_value[0]) - 1,
        'Annual Return': ((portfolio_value[-1] / portfolio_value[0]) ** (252/len(portfolio_value)) - 1),
        'Volatility': daily_returns.std() * (252 ** 0.5),
        'Sharpe Ratio': (daily_returns.mean() / daily_returns.std()) * (252 ** 0.5)
    }
    
    return pd.Series(metrics), portfolio_value

3. Market Analysis Report

Creating automated market analysis reports.

def generate_market_report(index_symbol='SPY', top_stocks=['AAPL', 'MSFT', 'GOOGL']):
    # Download market data
    market = yf.download(index_symbol, period='1y')
    
    # Calculate market metrics
    market_return = (market['Adj Close'][-1] / market['Adj Close'][0]) - 1
    market_vol = market['Adj Close'].pct_change().std() * (252 ** 0.5)
    
    # Analyze top stocks
    stock_metrics = {}
    for stock in top_stocks:
        data = yf.download(stock, period='1y')
        stock_return = (data['Adj Close'][-1] / data['Adj Close'][0]) - 1
        stock_metrics[stock] = {
            'Return': stock_return,
            'Relative Performance': stock_return - market_return
        }
    
    return {
        'Market Performance': market_return,
        'Market Volatility': market_vol,
        'Stock Analysis': pd.DataFrame(stock_metrics).T
    }
